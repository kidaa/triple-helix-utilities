001package org.triplehelix.lib;
002
003import edu.wpi.first.wpilibj.networktables2.util.List;
004
005/**
006 *
007 * @author robotics
008 */
009public class RollingAverager {
010    
011    private List values = new List();
012    private int numValues;
013    
014    /**
015     * Initialize an averager that will hold the specified number of values.
016     * @param numValues The number of entries the averager will store.
017     */
018    public RollingAverager(int numValues) {
019        if (numValues <= 0) {
020            throw new IllegalArgumentException("Cannot have 0 or fewer values.");
021        }
022        this.numValues = numValues;
023    }
024    
025    /**
026     * Initialize an averager that will hold the specified number of values and initialize the values to the specified value.
027     * @param numValues The number of entries the averager will store.
028     * @param initialValue The value to initialize the averager to.
029     */
030    public RollingAverager(int numValues, double initialValue) {
031        this(numValues);
032        for (int i = 0; i < numValues; i++) {
033            values.add(new Double(initialValue));
034        } 
035    }
036    
037    /**
038     * Add a new value to the averager.
039     * @param value The value to add to the averager.
040     */
041    public void addValue(double value) {
042        if (values.size() >= numValues) {
043            values.remove(0);
044        }
045        values.add(new Double(value));
046    }
047    
048    /**
049     * Get the current average of the averager.
050     * @return The current average.
051     */
052    public double getAverage() {
053        double sum = 0;
054        
055        for(int i = 0; i < values.size(); i++) {
056            sum += ((Double) values.get(i)).doubleValue();
057        }
058        return sum / values.size();
059    }
060}
