001package org.triplehelix.lib.buttons;
002
003import edu.wpi.first.wpilibj.DigitalInput;
004import edu.wpi.first.wpilibj.buttons.Button;
005
006/**
007 * A button that reads from a Digital Ipout on the robot.
008 * @author robotics
009 */
010public class DigitalInputButton extends Button {
011    
012    private DigitalInput input;
013    
014    /**
015     * Initialize a button that reads from the specified digital input port.
016     * @param port 
017     */
018    public DigitalInputButton(int port) {
019        input = new DigitalInput(port);
020    }
021
022    /**
023     * Get the current state of the digital input.
024     * @return The current state.
025     */
026    public boolean get() {
027        return input.get();
028    }
029    
030}
