001package org.triplehelix.lib.commands.autonomous;
002
003import edu.wpi.first.wpilibj.buttons.Button;
004import edu.wpi.first.wpilibj.buttons.Trigger;
005import edu.wpi.first.wpilibj.networktables2.util.List;
006import org.triplehelix.lib.properties.provides.IProvidesName;
007
008/**
009 * A binary selector using 3 buttons that will return the command that is currently selected.
010 * @author robotics
011 */
012public class AutonomousSelector extends BinaryConverter {
013    
014    private List commandList = new List();
015    
016    /**
017     * Initialize a new autonomous selector with the provided inputs.
018     * @param lsb Least significant bit
019     * @param mb Middle bit
020     * @param msb Most significant bit
021     */
022    public AutonomousSelector(Trigger lsb, Trigger mb, Trigger msb){
023        super(lsb, mb, msb);
024    }
025    
026    public void setCommands(List commandList){
027        this.commandList = commandList;
028    }
029
030    public IProvidesName getSelectedCommand(){
031        if (getSelectedNumber() >= commandList.size()) {
032            return new NoneCommand();
033        }
034        return (IProvidesName)commandList.get(this.getSelectedNumber());
035    }
036       
037}
