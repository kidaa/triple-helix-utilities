001package org.triplehelix.lib.controller;
002
003import edu.wpi.first.wpilibj.Joystick;
004
005/**
006 * XBox controller wrapper for a joystick.
007 * @author Matthew.Lythgoe
008 */
009public class XBoxController extends Joystick {
010    
011    public static final int A = 0;
012    public static final int B = 1;
013    public static final int X = 2;
014    public static final int Y = 3;
015    public static final int LEFT_BUMPER = 4;
016    public static final int RIGHT_BUMPER = 5;
017    public static final int BACK = 6;
018    public static final int START = 7;
019    public static final int LEFT_STICK = 8;
020    public static final int RIGHT_STICK = 9;
021    
022    public static final int LEFT_X = 1;
023    public static final int LEFT_Y = 2;
024    public static final int TRIGGERS = 3;
025    public static final int RIGHT_X = 4;
026    public static final int RIGHT_Y = 5;
027    public static final int D_PAD_X = 6;
028    public static final int D_PAD_Y = 7;
029    
030    public XBoxController(int port) {
031        super(port);
032    }
033 
034    /**
035     * Read the value of the right joystick's X axis.
036     * @return the value of the right joystick's X axis.
037     */
038    public double getRightStickX() {
039        return getRawAxis(RIGHT_X);
040    }
041
042    /**
043     * Read the value of the right joystick's Y axis.
044     * @return the value of the right joystick's Y axis.
045     */
046    public double getRightStickY() {
047        return getRawAxis(RIGHT_Y);
048    }
049
050    /**
051     * Read the value of the left joystick's X axis.
052     * @return the value of the left joystick's X axis.
053     */
054    public double getLeftStickX() {
055        return getRawAxis(LEFT_X);
056    }
057
058    /**
059     * Read the value of the left joystick's Y axis.
060     * @return the value of the left joystick's Y axis.
061     */
062    public double getLeftStickY() {
063        return getRawAxis(LEFT_Y);
064    }
065    
066    /**
067     * Read the value of the d-pad's X axis.
068     * @return the value of the d-pad's X axis.
069     */
070    public double getDPadX() {
071        return getRawAxis(D_PAD_X);
072    }
073    
074    /**
075     * Read the value of the d-pad's Y axis.
076     * @return the value of the d-pad's Y axis.
077     */
078    public double getDPadY() {
079        return getRawAxis(D_PAD_Y);
080    }
081
082    /**
083     * Read the state of the A button.
084     * @return the state of the A button.
085     */
086    public boolean getAButton() {
087        return getRawButton(A);
088    }
089
090    /**
091     * Read the state of the B button.
092     * @return the state of the B button.
093     */
094    public boolean getBButton() {
095        return getRawButton(B);
096    }
097
098    /**
099     * Read the state of the X button.
100     * @return the state of the X button.
101     */
102    public boolean getXButton() {
103        return getRawButton(X);
104    }
105
106    /**
107     * Read the state of the Y button.
108     * @return the state of the Y button.
109     */
110    public boolean getYButton() {
111        return getRawButton(Y);
112    }
113
114    /**
115     * Read the state of the back button.
116     * @return the state of the back button.
117     */
118    public boolean getBackButton() {
119        return getRawButton(BACK);
120    }
121
122    /**
123     * Read the state of the start button.
124     * @return the state of the start button.
125     */
126    public boolean getStartButton() {
127        return getRawButton(START);
128    }
129
130    /**
131     * Read the state of the right bumper button.
132     * @return the state of the right bumper button.
133     */
134    public boolean getRightBumperButton() {
135        return getRawButton(RIGHT_BUMPER);
136    }
137
138    /**
139     * Read the state of the left bumper button.
140     * @return the state of the left bumper button.
141     */
142    public boolean getLeftBumperButton() {
143        return getRawButton(LEFT_BUMPER);
144    }
145
146    /**
147     * Read the state of the left stick button.
148     * @return the state of the left stick button.
149     */
150    public boolean getLeftStickButton() {
151        return getRawButton(LEFT_STICK);
152    }
153
154    /**
155     * Read the state of the right stick button.
156     * @return the state of the right stick button.
157     */
158    public boolean getRightStickButton() {
159        return getRawButton(RIGHT_STICK);
160    } 
161        
162    /**
163     * Read the state of the right trigger.
164     * @return the state of the right trigger.
165     */
166    public double getRightTrigger() {
167        return -Math.min(getRawAxis(TRIGGERS), 0);
168    }
169
170    /**
171     * Read the state of the left trigger.
172     * @return the state of the left trigger.
173     */
174    public double getLeftTrigger() {
175        return Math.max(getRawAxis(TRIGGERS), 0);
176    }
177}
