001package org.triplehelix.lib.pid;
002
003import edu.wpi.first.wpilibj.PIDOutput;
004
005/**
006 * Holds the output from the PID controller for later retrieval.
007 * @author robotics
008 */
009public class GenericPIDOutput implements PIDOutput {
010    
011    private double power;
012
013    /**
014     * Store the calculated PID output.
015     * @param d The calculated PID output.
016     */
017    public void pidWrite(double d) {
018        power = d;
019    }
020    
021    /**
022     * Retrieve the latest calculated PID output.
023     * @return The latest calculated PIT output.
024     */
025    public double getOutput() {
026        return power;
027    }
028}
